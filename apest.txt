    <script>
        // --- ESTADO DA APLICAÇÃO ---
        let DB = {
            categorias: JSON.parse(localStorage.getItem('MGM_cats')) || [],
            produtos: JSON.parse(localStorage.getItem('MGM_prods')) || [],
            movimentacoes: JSON.parse(localStorage.getItem('MGM_movs')) || []
        };

        const saveDB = () => {
            localStorage.setItem('MGM_cats', JSON.stringify(DB.categorias));
            localStorage.setItem('MGM_prods', JSON.stringify(DB.produtos));
            localStorage.setItem('MGM_movs', JSON.stringify(DB.movimentacoes));
            renderAll();
        };

        // --- NAVEGAÇÃO ---
        function showTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            
            // Set current dates for forms
            const today = new Date().toISOString().split('T')[0];
            if(document.getElementById('ent-data')) document.getElementById('ent-data').value = today;
            if(document.getElementById('sai-data')) document.getElementById('sai-data').value = today;

            lucide.createIcons();
        }

        // --- CATEGORIAS ---
        function salvarCategoria() {
            const cat = document.getElementById('cat-nome').value.trim();
            const tipo = document.getElementById('cat-tipo').value.trim();
            if(!cat || !tipo) return alert("Preencha categoria e tipo!");
            
            DB.categorias.push({ cat, tipo });
            document.getElementById('cat-nome').value = '';
            document.getElementById('cat-tipo').value = '';
            saveDB();
        }

        function renderCategorias() {
            const container = document.getElementById('lista-categorias');
            const select = document.getElementById('p-categoria-completa');
            const audCat = document.getElementById('aud-cat');

            container.innerHTML = '';
            select.innerHTML = '<option value="">Selecione Categoria/Tipo</option>';
            audCat.innerHTML = '<option value="">Todas Categorias</option>';

            DB.categorias.forEach((c, index) => {
                const label = `${c.cat} - ${c.tipo}`;
                container.innerHTML += `<span class="bg-slate-200 px-3 py-1 rounded-full text-xs font-semibold flex items-center gap-2">
                    ${label} <button onclick="removerCat(${index})" class="text-red-500 hover:text-red-700">×</button>
                </span>`;
                select.innerHTML += `<option value="${label}">${label}</option>`;
                audCat.innerHTML += `<option value="${c.cat}">${c.cat}</option>`;
            });
        }

        function removerCat(idx) {
            DB.categorias.splice(idx, 1);
            saveDB();
        }

        // --- PRODUTOS ---
        document.getElementById('form-produto').onsubmit = (e) => {
            e.preventDefault();
            const novoProd = {
                id: document.getElementById('p-id').value,
                nome: document.getElementById('p-nome').value,
                dosagem: document.getElementById('p-dosagem').value,
                unidade: document.getElementById('p-unidade').value,
                volume: document.getElementById('p-volume').value,
                apresentacao: document.getElementById('p-apresentacao').value,
                categoria: document.getElementById('p-categoria-completa').value,
                minimo: Number(document.getElementById('p-minimo').value)
            };
            
            if(DB.produtos.find(p => p.id === novoProd.id)) return alert("Este ID já existe!");
            
            DB.produtos.push(novoProd);
            e.target.reset();
            saveDB();
            alert("Produto cadastrado com sucesso!");
        };

        // --- BUSCA DINÂMICA (ENTRADA/SAÍDA) ---
        function buscarProduto(contexto) {
            const query = document.getElementById(`search-${contexto}`).value.toLowerCase();
            const resultDiv = document.getElementById(`result-${contexto}`);
            const form = document.getElementById(`form-${contexto}`);

            if(query.length < 2) {
                form.classList.add('opacity-50', 'pointer-events-none');
                return;
            }

            const p = DB.produtos.find(prod => 
                prod.id.toLowerCase().includes(query) || 
                prod.nome.toLowerCase().includes(query)
            );

            if(p) {
                resultDiv.innerHTML = `<span class="text-green-600 font-bold">✓ Encontrado: ${p.nome} (${p.apresentacao})</span>`;
                form.classList.remove('opacity-50', 'pointer-events-none');
                
                if(contexto === 'entrada') {
                    document.getElementById('ent-id-base').value = p.id;
                    document.getElementById('info-ent-id').innerText = p.id;
                    document.getElementById('info-ent-cat').innerText = p.categoria;
                    document.getElementById('ent-atual-val').innerText = getSaldoTotal(p.id);
                } else {
                    document.getElementById('sai-id-base').value = p.id;
                    carregarLotesSaida(p.id);
                }
            } else {
                resultDiv.innerHTML = `<span class="text-red-500 italic">Produto não localizado.</span>`;
                form.classList.add('opacity-50', 'pointer-events-none');
            }
        }

        // --- LÓGICA DE ESTOQUE (SALDOS) ---
        function getSaldosPorLote(prodId) {
            const lotes = {};
            DB.movimentacoes.filter(m => m.prodId === prodId).forEach(m => {
                const key = `${m.lote}|${m.validade}`;
                if(!lotes[key]) lotes[key] = 0;
                lotes[key] += (m.tipo === 'ENTRADA' ? m.qtd : -m.qtd);
            });
            return lotes;
        }

        function getSaldoTotal(prodId) {
            const saldos = getSaldosPorLote(prodId);
            return Object.values(saldos).reduce((a, b) => a + b, 0);
        }

        // --- ENTRADA ---
        document.getElementById('form-entrada').onsubmit = (e) => {
            e.preventDefault();
            const mov = {
                tipo: 'ENTRADA',
                prodId: document.getElementById('ent-id-base').value,
                lote: document.getElementById('ent-lote').value,
                validade: document.getElementById('ent-validade').value,
                dataMov: document.getElementById('ent-data').value,
                qtd: Number(document.getElementById('ent-qtd').value)
            };
            DB.movimentacoes.push(mov);
            saveDB();
            alert("Entrada registrada!");
            e.target.reset();
            document.getElementById('search-entrada').value = '';
            buscarProduto('entrada');
        };

        // --- SAÍDA ---
        function carregarLotesSaida(prodId) {
            const select = document.getElementById('sai-lote-select');
            const saldos = getSaldosPorLote(prodId);
            select.innerHTML = '<option value="">Selecione o Lote (FIFO sugerido)</option>';
            
            Object.keys(saldos).forEach(key => {
                if(saldos[key] > 0) {
                    const [lote, valid] = key.split('|');
                    select.innerHTML += `<option value="${key}">${lote} (Val: ${valid}) - Saldo: ${saldos[key]}</option>`;
                }
            });
        }

        function updateEstoqueSaida() {
            const val = document.getElementById('sai-lote-select').value;
            const pId = document.getElementById('sai-id-base').value;
            if(!val) return;
            const saldos = getSaldosPorLote(pId);
            document.getElementById('sai-atual-val').innerText = saldos[val];
        }

        document.getElementById('form-saida').onsubmit = (e) => {
            e.preventDefault();
            const keyLote = document.getElementById('sai-lote-select').value;
            const qtdSaida = Number(document.getElementById('sai-qtd').value);
            const pId = document.getElementById('sai-id-base').value;
            const [lote, validade] = keyLote.split('|');

            const saldoLote = getSaldosPorLote(pId)[keyLote];

            if(qtdSaida > saldoLote) {
                return alert("ERRO: Quantidade de saída maior que o estoque deste lote! Operação bloqueada.");
            }

            DB.movimentacoes.push({
                tipo: 'SAIDA',
                prodId: pId,
                lote: lote,
                validade: validade,
                dataMov: document.getElementById('sai-data').value,
                qtd: qtdSaida
            });

            saveDB();
            alert("Saída registrada com sucesso!");
            e.target.reset();
            document.getElementById('search-saida').value = '';
            buscarProduto('saida');
        };

        // --- RENDERIZAR ESTOQUE ATUAL ---
        function renderEstoque() {
            const corpo = document.getElementById('lista-estoque');
            corpo.innerHTML = '';
            
            DB.produtos.forEach(p => {
                const saldos = getSaldosPorLote(p.id);
                Object.keys(saldos).forEach(key => {
                    if(saldos[key] > 0) {
                        const [lote, validade] = key.split('|');
                        corpo.innerHTML += `
                            <tr class="hover:bg-slate-50 border-b transition">
                                <td class="p-3 text-xs font-mono">${p.id}</td>
                                <td class="p-3 font-semibold">${p.nome}</td>
                                <td class="p-3">${lote}</td>
                                <td class="p-3 text-sm">${validade}</td>
                                <td class="p-3 font-bold text-blue-700">${saldos[key]} ${p.unidade}</td>
                            </tr>
                        `;
                    }
                });
            });
        }

        // --- DASHBOARD DE RISCOS ---
        function renderDashboard() {
            let totalItens = 0;
            let riscoMinimo = 0;
            let riscoValidade = 0;
            const hoje = new Date();
            const limiteValidade = new Date();
            limiteValidade.setDate(hoje.getDate() + 30);

            DB.produtos.forEach(p => {
                const saldoTotal = getSaldoTotal(p.id);
                if(saldoTotal > 0) totalItens++;
                if(saldoTotal > 0 && saldoTotal <= p.minimo) riscoMinimo++;

                const saldos = getSaldosPorLote(p.id);
                Object.keys(saldos).forEach(key => {
                    const [lote, validStr] = key.split('|');
                    const validade = new Date(validStr);
                    if(saldos[key] > 0 && validade <= limiteValidade) riscoValidade++;
                });
            });

            document.getElementById('dash-total').innerText = totalItens;
            document.getElementById('dash-minimo').innerText = riscoMinimo;
            document.getElementById('dash-validade').innerText = riscoValidade;
        }

        // --- AUDITORIA ---
        function gerarAuditoria() {
            const inicio = document.getElementById('aud-inicio').value;
            const fim = document.getElementById('aud-fim').value;
            const catFiltro = document.getElementById('aud-cat').value;
            const corpo = document.getElementById('tabela-aud-corpo');
            
            corpo.innerHTML = '';

            const filtrados = DB.movimentacoes.filter(m => {
                const p = DB.produtos.find(prod => prod.id === m.prodId);
                const dataM = m.dataMov;
                const matchData = (!inicio || dataM >= inicio) && (!fim || dataM <= fim);
                const matchCat = !catFiltro || (p && p.categoria.includes(catFiltro));
                return matchData && matchCat;
            });

            filtrados.reverse().forEach(m => {
                const p = DB.produtos.find(prod => prod.id === m.prodId);
                corpo.innerHTML += `
                    <tr>
                        <td class="p-2 border">${m.dataMov}</td>
                        <td class="p-2 border font-bold ${m.tipo === 'ENTRADA' ? 'text-green-600' : 'text-red-600'}">${m.tipo}</td>
                        <td class="p-2 border">${p ? p.nome : 'Excluído'}</td>
                        <td class="p-2 border text-xs">${m.lote}</td>
                        <td class="p-2 border font-mono">${m.qtd}</td>
                    </tr>
                `;
            });
        }

        // --- IMPORTAÇÃO EXCEL ---
        document.getElementById('importExcel').addEventListener('change', (e) => {
            const file = e.target.files[0];
            const reader = new FileReader();
            reader.onload = (evt) => {
                const data = evt.target.result;
                const workbook = XLSX.read(data, { type: 'binary' });
                const sheetName = workbook.SheetNames[0];
                const rows = XLSX.utils.sheet_to_json(workbook.Sheets[sheetName]);
                
                rows.forEach(row => {
                    // Mapeamento simples baseado nas colunas
                    if(row.ID && row.NOME) {
                        DB.produtos.push({
                            id: String(row.ID),
                            nome: row.NOME,
                            dosagem: row.DOSAGEM || '',
                            unidade: row.UNIDADE || 'Unid',
                            volume: row.VOLUME || '',
                            apresentacao: row.APRESENTACAO || '',
                            categoria: row.CATEGORIA || '',
                            minimo: Number(row.MINIMO) || 0
                        });
                    }
                });
                saveDB();
                alert(`${rows.length} produtos importados!`);
            };
            reader.readAsBinaryString(file);
        });
